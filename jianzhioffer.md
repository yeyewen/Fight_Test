```python
class Solution(object):
    def getMinimumDifference(self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        prev = None
        min_dif = float('inf')
        stack = []
        while stack != [] or root != None:
            while root:
                stack.append(root)
                root = root.left
            node = stack.pop()
            if prev != None:
                dif = node.val - prev.val
                if dif < min_dif:
                    min_dif = dif
            prev = node
            root = node.right
        return min_dif

```

```python
class Solution(object):

    def convertBST(self, root):
        if root is not None:

            root.val = self.total
```

### 面试题60.n个骰子的点数

n=5,6,7....
以此类推
循环结束就得到了答案。
```
class Solution:
    def twoSum(self, n: int) -> List[float]:
        dp=[1/6]*6
        for i in range(1,n):
            tmp=[0]*(5*i+6)
            for j in range(len(dp)):
                for k in range(6):
                    tmp[j+k]+= dp[j]*1/6 
            dp=tmp
        return dp

```

```
            if nums[m] < target: i = m + 1
            else: j = m - 1
        left = j
        return right - left - 1
```

```
class Solution:

        repeat = set()
        ma, mi = 0, 14

            if num == 0: 
                continue # 跳过大小王

        return ma - mi < 5 # 最大牌 - 最小牌 < 5 则可构成顺子
```

```
        if not str: return 0 # 字符串为空则直接返回
        res, i, sign = 0, 1, 1
        int_max, int_min, bndry = 2 ** 31 - 1, -2 ** 31, 2 ** 31 // 10

            if not '0' <= c <= '9' : break # 遇到非数字的字符则跳出
            if res > bndry or res == bndry and c > '7': return int_max if sign == 1 else int_min # 数字越界处理

        return sign * res
```

```
        if x == 0: return 0
        res = 1
        if n < 0: x, n = 1 / x, -n
        while n:
            if n & 1: res *= x
            x *= x
            n >>= 1
```

### 面试题41数据流的中位数
```
from heapq import *

class MedianFinder:
    def __init__(self):
        self.A = [] # 小顶堆，保存较大的一半
        self.B = [] # 大顶堆，保存较小的一半

    def addNum(self, num: int) -> None:
        if len(self.A) != len(self.B):
            heappush(self.A, num)
            heappush(self.B, -heappop(self.A))
        else:
            heappush(self.B, -num)
            heappush(self.A, -heappop(self.B))

    def findMedian(self) -> float:
        return self.A[0] if len(self.A) != len(self.B) else (self.A[0] - self.B[0]) / 2.0
```
